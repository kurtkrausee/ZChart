import { SceneNode } from './SceneNode';
import { TimeScale } from '../math/TimeScale';
import { PriceScale } from '../math/PriceScale';
import type { ChartConfig } from '../core/ChartOptions';
import { distanceToLineSegment } from '../utils/geometry';

export interface LogicalPoint {
  index: number;
  price: number;
}

export class TrendLineNode extends SceneNode {  // ← extends SceneNode neu

  public point1: LogicalPoint | null = null;
  public point2: LogicalPoint | null = null;
  public isHovered: boolean = false;
  public isSelected: boolean = false;

  // NEU: Label für Object Tree
  public label: string = 'Trendlinie';

  private readonly HIT_TOLERANCE = 5;

  // NEU: welcher Anfasser wird gerade gezogen? null | 'p1' | 'p2'
  public draggingAnchor: 'p1' | 'p2' | 'line' | null = null;

  public draw(
    ctx: CanvasRenderingContext2D,
    timeScale: TimeScale,
    priceScale: PriceScale,
    options: ChartConfig
  ): void {
    if (!this.point1 || !this.point2) return;

    const x1 = timeScale.indexToX(this.point1.index);
    const y1 = priceScale.priceToY(this.point1.price);
    const x2 = timeScale.indexToX(this.point2.index);
    const y2 = priceScale.priceToY(this.point2.price);

    ctx.save();
    ctx.strokeStyle = this.isSelected
      ? '#FFD700'
      : this.isHovered ? '#4ea2ff' : '#2962FF';
    ctx.lineWidth = this.isSelected ? 3 : 2;
    ctx.lineCap = 'round';

    ctx.beginPath();
    ctx.moveTo(x1, y1);
    ctx.lineTo(x2, y2);
    ctx.stroke();

    if (this.isSelected || this.isHovered) {
      this.drawAnchor(ctx, x1, y1, ctx.strokeStyle);
      this.drawAnchor(ctx, x2, y2, ctx.strokeStyle);
    }

    ctx.restore();
  }

  private drawAnchor(
    ctx: CanvasRenderingContext2D,
    x: number,
    y: number,
    color: string | CanvasGradient | CanvasPattern
  ): void {
    ctx.beginPath();
    ctx.arc(x, y, 4, 0, 2 * Math.PI);
    ctx.fillStyle = '#ffffff';
    ctx.fill();
    ctx.strokeStyle = color;
    ctx.stroke();
  }

  public hitTest(
    pixelX: number,
    pixelY: number,
    timeScale: TimeScale,
    priceScale: PriceScale
  ): boolean {
    if (!this.point1 || !this.point2) return false;

    const x1 = timeScale.indexToX(this.point1.index);
    const y1 = priceScale.priceToY(this.point1.price);
    const x2 = timeScale.indexToX(this.point2.index);
    const y2 = priceScale.priceToY(this.point2.price);

    return distanceToLineSegment(pixelX, pixelY, x1, y1, x2, y2) <= this.HIT_TOLERANCE;
  }

  // NEU: welcher Anfasser liegt unter dem Mauszeiger?
  // Gibt null zurück wenn keiner getroffen, sonst 'p1' | 'p2'
  public hitTestAnchor(
    pixelX: number,
    pixelY: number,
    timeScale: TimeScale,
    priceScale: PriceScale
  ): 'p1' | 'p2' | null {
    if (!this.point1 || !this.point2) return null;

    const x1 = timeScale.indexToX(this.point1.index);
    const y1 = priceScale.priceToY(this.point1.price);
    const x2 = timeScale.indexToX(this.point2.index);
    const y2 = priceScale.priceToY(this.point2.price);

    const ANCHOR_RADIUS = 8;  // etwas grosszügiger als Linie
    const d1 = Math.hypot(pixelX - x1, pixelY - y1);
    const d2 = Math.hypot(pixelX - x2, pixelY - y2);

    if (d1 <= ANCHOR_RADIUS) return 'p1';
    if (d2 <= ANCHOR_RADIUS) return 'p2';
    return null;
  }

  // NEU: Persistenz — Linie als JSON speichern
  public serialize(): object {
    return {
      type:   this.getType(),
      id:     this.id,
      label:  this.label,
      zIndex: this.zIndex,
      point1: this.point1,
      point2: this.point2,
      style: {
        color:     '#2962FF',
        lineWidth: 2
      }
    };
  }

  // NEU: Wiederherstellen aus gespeichertem JSON
  public static deserialize(data: any): TrendLineNode {
    const node = new TrendLineNode();
    node.id     = data.id;
    node.label  = data.label ?? 'Trendlinie';
    node.zIndex = data.zIndex ?? 0;
    node.point1 = data.point1;
    node.point2 = data.point2;
    return node;
  }

  public getType(): string {
    return 'trendline';
  }
}
